/* Given a non-empty array of integers nums, every element appears twice except for one. Find that single one.
Sample :- Input: nums = [2,2,1]
Output: 1
Note: Take all the required input from user. */

#include<iostream>
using namespace std;

int main(){
    int arr[10];
    int n;
    cin >> n;
    for(int i=0;i<n;i++){
        cin >> arr[i];
    }
    int counter,i;
    cout << "The number occuring only once is: " << endl;
    for(int i=0;i<n;i++){
        counter=0;
        for(int j=0;j<n;j++){
            if(arr[j] == arr[i])
            ++counter;
        }
        if(counter != 2){
        cout << arr[i] << " " << endl;
        }
    }
}
